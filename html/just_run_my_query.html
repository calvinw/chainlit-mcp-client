    subquery.`Net Revenue 2023`,

    fm.`Current_Ratio` AS `Current Ratio`,

    fm.`Debt_to_Equity` AS `Debt to Equity`
FROM (

        c.subsegment

        new_company_info c ON f2024.company_name = c.company
    WHERE
        f2024.year = 2024
) AS subquery
JOIN
